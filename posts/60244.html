void quick_sort(int a[],int L,int R){
        if(L > R){
            return;
        }
        int left = L;
        int right = R;
        int key = a[left];
        while(left < right){
            while(left < right && a[right] >= key ){
                right--;
            }
            if(left < right){
                a[left] = a[right];
            }
            while(left < right && a[left] <= key){
                left++;
            }
            if(left < right){
                a[right] = a[left];
            }
            if(left >= right){
                a[left] = key;
            }
        }
        quick_sort(a, L,right - 1);
        quick_sort(a, right + 1,R);
    }
