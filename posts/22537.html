class Solution {
public:
    int FindGreatestSumOfSubArray(vector<int> array) {
        if(array.size() <= 0)
            return 0;
        int temp = 0;
        int max = array[0];
        for(int i = 0;i < array.size();i++){
            if(temp <= 0)
                temp = array[i];
            else{
                temp += array[i];
            }

            if(temp > max){
                max = temp;
            }
        }
        return max;
    }
};
