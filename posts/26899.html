MainActiviity.java代码

各功能已写好注释，选项菜单中有子菜单，上下文菜单也写好了。

package com.gdlgxy.menudemo;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.ContextMenu;
import android.view.Menu;
import android.view.MenuInflater;
import android.view.MenuItem;
import android.view.SubMenu;
import android.view.View;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        //声明textview控件，注册上下文菜单
        TextView textView = (TextView) findViewById(R.id.context_menu);
        registerForContextMenu(textView);
    }

    //选项菜单中创建两个子菜单，分别是泰罗奥特曼和梦比优斯奥特曼
    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        SubMenu s1=menu.addSubMenu(1, 0, 1, "泰罗奥特曼");
        s1.setHeaderTitle("技能");//设置子菜单标题
        s1.add(1, 1, 1, "斯特流姆光线");
        s1.add(1, 2, 1, "箭形光线");
        s1.add(1, 3, 1, "泰罗大火焰");
        s1.add(1, 4, 1, "禁术：奥特炸弹");

        SubMenu s2=menu.addSubMenu(2, 0, 1, "梦比优斯奥特曼");
        s2.setHeaderTitle("技能");
        s2.add(2, 1, 1, "梦比姆射线");
        s2.add(2, 2, 1, "梦比姆光剑");
        s2.add(2, 3, 1, "梦比姆爆裂");
        s2.add(2, 4, 1, "梦比姆回旋飞踢");

        return super.onCreateOptionsMenu(menu);
    }

    //设置点击事件，弹出信息给人类
    @Override
    public boolean onOptionsItemSelected(MenuItem item) {
        if(item.getGroupId()==1){//判断分组
            switch(item.getItemId()){
                case 1:
                    Toast.makeText(MainActivity.this, "释放斯特流姆光线", Toast.LENGTH_SHORT).show();
                    break;
                case 2:
                    Toast.makeText(MainActivity.this, "释放箭形光线", Toast.LENGTH_SHORT).show();
                    break;
                case 3:
                    Toast.makeText(MainActivity.this, "释放泰罗大火焰", Toast.LENGTH_SHORT).show();
                    break;
                case 4:
                    Toast.makeText(MainActivity.this, "赌上性命的战斗，奥特炸弹", Toast.LENGTH_SHORT).show();
                    break;
            }
        }else if(item.getGroupId()==2){
            switch(item.getItemId()){
                case 1:
                    Toast.makeText(MainActivity.this, "释放梦比姆光线", Toast.LENGTH_SHORT).show();
                    break;
                case 2:
                    Toast.makeText(MainActivity.this, "使用梦比姆光剑", Toast.LENGTH_SHORT).show();
                    break;
                case 3:
                    Toast.makeText(MainActivity.this, "释放梦比姆爆裂", Toast.LENGTH_SHORT).show();
                    break;
                case 4:
                    Toast.makeText(MainActivity.this, "释放梦比姆回旋飞踢", Toast.LENGTH_SHORT).show();
                    break;
            }
        }
        return super.onOptionsItemSelected(item);
    }

    //上下文菜单
    @Override
    public void onCreateContextMenu(ContextMenu menu, View v, ContextMenu.ContextMenuInfo menuInfo) {
        // TODO Auto-generated method stub
        super.onCreateContextMenu(menu, v, menuInfo);
        /*
         * 添加菜单项xml文件
         */
        MenuInflater inflater=new MenuInflater(this);
        inflater.inflate(R.menu.contextmenu, menu);

    }

    /*
     * ContextMenu事件处理
     */
    @Override
    public boolean onContextItemSelected(MenuItem item) {
        // TODO Auto-generated method stub
        switch (item.getItemId()) {
            case R.id.menu1:
                Toast.makeText(this, "初代与你并肩作战", Toast.LENGTH_SHORT).show();
                break;
            case R.id.menu2:
                Toast.makeText(this, "赛文与你并肩作战", Toast.LENGTH_SHORT).show();
                break;
            case R.id.menu3:
                Toast.makeText(this, "杰克与你并肩作战", Toast.LENGTH_SHORT).show();
                break;
            case R.id.menu4:
                Toast.makeText(this, "艾斯与你并肩作战", Toast.LENGTH_SHORT).show();
                break;
            default:
                break;
        }
        return super.onContextItemSelected(item);
    }

    /*
     * 测试事件
     */
    private void Toastmake(int itemid) {
        Toast.makeText(MainActivity.this, "你点击了上下文菜单"+itemid, Toast.LENGTH_SHORT).show();
    }

}
