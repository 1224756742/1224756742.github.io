输入

先让两个正整数等长，短的补零。模拟加法运算，从尾部开始，若产生进位就在前一位加1。

#include<iostream>
#include<string>
using namespace std;

int main(){
    string str1;
    string str2;
    while(cin>>str1>>str2){
        while(str1.size() < str2.size()){
            str1 = '0' + str1;
        }
        while(str1.size() > str2.size()){
            str2 = '0' + str2;
        }
        int temp = 0,carry = 0;
        for(int i = str1.size() - 1;i >= 0;i--){
            temp = str1[i] - '0' + str2[i] -'0' + carry;
            str1[i] = temp % 10 + '0';
            if(temp / 10){
                carry = 1;
            }else{
                carry = 0;
            }
        }
        if(carry){
            str1 = "1" + str1;
        }
        cout<<str1<<endl;
    }
    return 0;
}
